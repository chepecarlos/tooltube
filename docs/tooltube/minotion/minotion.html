 24def consultaPost(ruta: str):
 25    """Consulta la API de Notion para obtener información sobre una ruta específica.
 26
 27    Args:
 28        ruta (str): La ruta del archivo a consultar en Notion.
 29
 30    Returns:
 31        dict: La respuesta de la API de Notion o None si no se encontró información.
 32    """
 33    
 34    
 35    dataNotion = miLibrerias.ObtenerArchivo("data/notion.md")
 36    
 37    if dataNotion is None:
 38        logger.warning("No data de Notion")
 39        return None
 40    urlConsulta = f"https://api.notion.com/v1/databases/{dataNotion.get('base_datos')}/query"
 41    cabezaConsulta = {
 42        "Authorization": f"Bearer {dataNotion.get('token')}",
 43        "Content-Type": "application/json",
 44        "Notion-Version": "2021-08-16"
 45    }
 46
 47    busqueda = {
 48        "filter": {
 49            "property": "Ruta Archivo",
 50            "text": {
 51                "contains": ruta
 52            }
 53        }
 54    }
 55
 56    respuesta = requests.post(urlConsulta, headers=cabezaConsulta, data=json.dumps(busqueda))
 57    if respuesta.status_code == 200:
 58        dataRespuesta = respuesta.json()
 59        dataRespuesta = dataRespuesta.get("results")
 60        if len(dataRespuesta) == 0:
 61            logger.warning("No se encontró en Notion")
 62            return None
 63        dataRespuesta = dataRespuesta[0]
 64        return dataRespuesta
