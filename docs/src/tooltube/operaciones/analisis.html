 70def convertir_fecha(fecha_str):
 71    global mes_actual
 72    global año_actual
 73    locale.setlocale(locale.LC_TIME, 'es_SV.utf8')  # Linux/macOS
 74
 75    partes = fecha_str.split(" de ")
 76    dia = int(partes[0])
 77    mes_nombre = partes[1]
 78
 79    # Convertir nombre del mes a número
 80    mes_numero = pd.to_datetime(mes_nombre, format='%B').month
 81
 82    # Manejo de transición de diciembre a enero
 83    if mes_numero == 1 and mes_actual == 12:
 84        año_final = año_actual + 1  # Estamos en diciembre y la fecha es enero -> año siguiente
 85    elif mes_numero == 12 and mes_actual == 1:
 86        año_final = año_actual - 1  # Estamos en enero y la fecha es diciembre -> año anterior
 87        print("Anterior ano")
 88    else:
 89        año_final = año_actual  # Mantener el mismo año en meses normales
 90    mes_actual = mes_numero
 91    año_actual = año_final
 92
 93    # Crear fecha con el año actual
 94    return pd.Timestamp(year=año_final, month=mes_numero, day=dia)

315def encontrarMaxMin(valores):
316    max30 = valores[0]
317    min30 = valores[0]
318    for valor in valores:
319        if min30 < valor:
320            min30 = valor
321        if max30 > valor:
322            max30 = valor
323    return (min30, max30)

326def encontrarSumas(valores):
327    sum7 = []
328    sum30 = []
329    suma365 = []
330    cantidad = len(valores)
331    for id in range(len(valores)):
332        sum7.append(0)
333        sum30.append(0)
334        suma365.append(0)
335        for j in range(id - 6, id + 1):
336            if j >= 0 and j < cantidad:
337                sum7[id] += valores.iloc[j]
338        sum7[id] /= 7
339        for j in range(id - 29, id + 1):
340            if j >= 0 and j < cantidad:
341                sum30[id] += valores.iloc[j]
342        sum30[id] /= 30
343        for j in range(id - 364, id + 1):
344            if j >= 0 and j < cantidad:
345                suma365[id] += valores.iloc[j]
346        suma365[id] /= 365
347    return (sum7, sum30, suma365)

350def tiempoASegundos(tiempo):
351    segundos = 0
352    if isinstance(tiempo, str):
353        pedadosTiempo = tiempo.split(":")
354        segundos = int(pedadosTiempo[-1])
355        if len(pedadosTiempo) > 1:
356            segundos += int(pedadosTiempo[-2]) * 60
357        if len(pedadosTiempo) > 2:
358            segundos += int(pedadosTiempo[-3]) * 3600
359    return segundos
